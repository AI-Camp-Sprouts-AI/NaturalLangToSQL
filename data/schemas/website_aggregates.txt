CREATE TABLE website_aggregates (
    id SERIAL PRIMARY KEY,
    dt DATE,
    customer_domain VARCHAR(255),
    lead_domain VARCHAR(255),
    ip_country VARCHAR(255),
    no_of_visiting_ips BIGINT,
    no_of_hits BIGINT,
    lead_domain_name VARCHAR(255),
    industry VARCHAR(255),
    estimated_num_employees INT,
    city VARCHAR(255),
    state VARCHAR(255),
    company_country VARCHAR(255),
    annual_revenue FLOAT,
    total_funding FLOAT,
    latest_funding_stage VARCHAR(255),
    status VARCHAR(255),
    decayed_inbound_score DOUBLE,
    decayed_intent_score DOUBLE,
    decayed_clubbed_score DOUBLE,
    last_visit_date DATE,
    employee_range VARCHAR(255),
    revenue_range VARCHAR(255)
);


GUIDELINES:
1. if just domain is mentioned, always compare it with customer_domain
2. if 'lead' is mentioned before domain name, compare it lead_domain
2. if country name is abbreviated, use full name of the country
3. industry name should always be in lowercase
4. visitors and users can be used interchangeably 
5. For total count of visitors or users, you have to add the number of visiting ips, don't count the rows or no of hits
6. For employee ranges always use estimated number of employees to compare
6. For revenue ranges always use annual revenue to compare, don't use revenue_range
